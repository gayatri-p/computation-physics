    linkcolor=Tue-red,
    urlcolor=Tue-red,
    citecolor = Tue-red
    }

\renewcommand{\labelenumii}{\textbf{\textcolor{Tue-red}{[\alph*]}}}% level 2
\renewcommand{\labelenumiii}{\textbf{\textcolor{Tue-red}{\roman*.}}}% level 3
